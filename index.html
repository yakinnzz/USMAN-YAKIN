1. PENGERTIAN OOP

Object Oriented Programming (OOP) adalah paradigma pemrograman yang menggunakan objek sebagai dasar perancangan program. Pendekatan ini memungkinkan kode menjadi lebih terstruktur, modular, dan mudah dikembangkan.

2. Class Dan Object ✓ Sama dengan DartPad

Class merupakan blueprint atau cetak biru yang digunakan untuk membuat objek. Class berisi atribut (variabel) yang menggambarkan karakteristik suatu objek serta method (fungsi) yang menggambarkan perilaku objek tersebut. Sedangkan object merupakan instansi dari sebuah class.

// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006
// Kelas  : Si2A

class Car {
  String? model;
  String? color;
  
  void start() {
    print('Mobil dinyalakan');
  }
  
  void stop() {
    print('Mobil dimatikan');
  }
}

void main() {
  Car myCar = Car();
  myCar.model = 'Toyota';
  myCar.color = 'Merah';
  
  print('Model: ${myCar.model}');
  print('Warna: ${myCar.color}');
  myCar.start();
  myCar.stop();
}

4. VALIDASI DATA ✓ Sama dengan DartPad

Validasi data dalam constructor memastikan data yang dimasukkan sesuai aturan.

// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006
// Kelas  : Si2A

class Mahasiswa {
  String nama;
  String nim;
  double ipk;

  // Constructor dengan validasi
  Mahasiswa(this.nama, this.nim, this.ipk) {
    if (ipk < 0.0 || ipk > 4.0) {
      throw ArgumentError('IPK harus antara 0.0 - 4.0');
    }
    if (nim.isEmpty) {
      throw ArgumentError('NIM tidak boleh kosong');
    }
  }

  void tampilkanData() {
    print('Nama : $nama');
    print('NIM  : $nim');
    print('IPK  : $ipk');
  }
}

void main() {
  try {
    // Data valid
    Mahasiswa mhs1 = Mahasiswa("USMAN YAKIN TRIO SINADA", "251410006", 3.90);
    print('=== DATA VALID ===');
    mhs1.tampilkanData();
    
    // Data tidak valid (akan error)
    print('\n=== Mencoba data tidak valid ===');
    Mahasiswa mhs2 = Mahasiswa("Test", "", 5.0);
    
  } catch (e) {
    print('Error: $e');
  }
}

5. ENCAPSULATION ✓ Sama dengan DartPad

Encapsulation membungkus data dan method serta membatasi akses langsung dari luar class.

// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006
// Kelas  : Si2A

class Person {
  String _name;  // Private attribute
  int age;       // Public attribute

  // Constructor
  Person(this._name, this.age);

  // Setter untuk mengubah _name
  void setName(String name) {
    if (name.isNotEmpty) {
      _name = name;
    } else {
      print('Nama tidak boleh kosong');
    }
  }

  // Getter untuk mengambil _name
  String getName() => _name;
  
  // Method untuk menampilkan data
  void displayInfo() {
    print('Nama: $_name, Umur: $age');
  }
}

void main() {
  // Membuat objek
  Person person = Person("USMAN YAKIN TRIO SINADA", 18);
  
  print('=== DATA AWAL ===');
  person.displayInfo();
  
  // Mengubah nama melalui setter
  print('\n=== SETELAH DIUBAH ===');
  person.setName("USMAN");
  person.displayInfo();
  
  // Mengakses melalui getter
  print('\n=== MENGGUNAKAN GETTER ===');
  print('Nama dari getter: ${person.getName()}');
  print('Umur (public): ${person.age}');
}

6. NAMED PARAMETER ✓ Sama dengan DartPad

Named parameter memungkinkan parameter dipanggil berdasarkan nama, membuat kode lebih jelas.

// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006
// Kelas  : Si2A

class Product {
  String nama;
  double harga;
  int stok;
  String kategori;

  // Constructor dengan named parameters
  Product({
    required this.nama,
    required this.harga,
    this.stok = 0,
    this.kategori = 'Umum',
  });

  void displayInfo() {
    print('''
    Nama Produk : $nama
    Harga      : Rp${harga.toStringAsFixed(0)}
    Stok       : $stok
    Kategori   : $kategori
    ''');
  }
}

void main() {
  // Membuat produk dengan named parameters
  Product produk1 = Product(
    nama: "Usman Website", 
    harga: 1500000,
    stok: 10,
    kategori: "Digital"
  );
  
  Product produk2 = Product(
    nama: "Cyber Security Course",
    harga: 2500000,
    // stok menggunakan default 0
    // kategori menggunakan default 'Umum'
  );
  
  print('=== PRODUK 1 ===');
  produk1.displayInfo();
  
  print('=== PRODUK 2 (dengan default value) ===');
  produk2.displayInfo();
}
